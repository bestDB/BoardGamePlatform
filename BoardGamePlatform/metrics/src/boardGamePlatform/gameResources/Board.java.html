11  /**
12   * Klasa reprezentujaca plansze na ktorej odbywa sie rozgrywka.
13   */
14  public class Board implements Serializable{
15      
16      /** Pola planszy. */
17      private List<Field> fields;
18      
19      /**
20       * Instancjonuje nowa plansze.
21       */
22      public Board() {
23          fields = new LinkedList<Field>();
24      }
25      
26      /**
27       * Instancjonuje nowa plansze.
28       *
29       * @param fields lista pol 
30       */
31      public Board( List<Field> fields) {
32          this.fields = fields;
33      }
34      
35      /**
36       * Zwraca liste pol z ktorych sklada sie plansza.
37       *
38       * @return lista pol
39       */
40      public List<Field> getFields() {
41          return fields;
42      }
43      
44      /**
45       * Ustawia pola z ktorych sklada sie plansza.
46       *
47       * @param fields lista pol
48       */
49      public void setFields(List<Field> fields) {
50          this.fields = fields;
51      }
52      
53      public String toString() {
54          String boardState = new String();
55          boardState += "Board:\n";
56          for(Field f : fields) {
57              boardState += f.toString("\t") + "\n";
58          }
59          boardState += "EndBoard\n";
60          return boardState;
61      }
