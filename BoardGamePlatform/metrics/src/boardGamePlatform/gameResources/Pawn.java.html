135     public String toString(String tabs) {
136         String pawnState = new String();
137         pawnState += tabs + "Pawn\n";
138         pawnState += tabs + tabs + Integer.toString(pawnID) + "\n";
139         pawnState += tabs +tabs + "Items:\n";
140         for(String i : pawnItems.keySet()) {
141             pawnState += tabs + i + " " + pawnItems.get(i).toString() + "\n";
142         }
143         pawnState += tabs +tabs + "EndItems\n";
144         pawnState += tabs + "EndPawn";
145         return pawnState;
