16  /**
17   * Klasa ulatwiajaca tworzenie nowej planszy.
18   */
19  public class BoardBuilder {
20      
21      /** Plansza. */
22      private Board board;
23      
24      /**
25       * Instancjonuje nowa klase pomocnicza.
26       *
27       * @param board obiekt planszy na ktorym ma operowac klasa pomocnicza
28       */
29      public BoardBuilder(Board board) {
30          this.board = board;
31      }
32      
33      /**
34       * Tworzy plansze z jedna mozliwa sciezka.
35       *
36       * @param size ilosc pol planszy
37       * @throws UnsupportedBoardSizeException podany zostal nieprawidlowy rozmiar planszy ( size <= 1)
38       */
39      public void makeSimpleLinearBoard( int size ) throws UnsupportedBoardSizeException {
40          if(size < 2)
41              throw new UnsupportedBoardSizeException();
42      
43          List<Field> fields = new LinkedList<Field>();
44          
45          for(int i = 0; i<size; i++) {
46              fields.add(new Field(i));
47          }
48          
49          for(int i = 0; i<size; i++) {
50              fields.get(i).addNeighbourField(fields.get((i+1) % size));
51          }
52          
53          board.setFields(fields);
54      }
55  }
