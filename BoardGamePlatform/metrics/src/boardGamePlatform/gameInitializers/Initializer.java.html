75      /**
76       * Initialize.
77       *
78       * @param playersCount the players count
79       * @return true, if successful
80       */ 
81      public final boolean initialize(int playersCount) {
82          if(gameDetails.ifPlayersCountPossible(playersCount)) {
83              Map<String, Item> items = itemsInitializer.initializeItems(playersCount);
84              Map<String, Pawn> pawns = pawnsInitializer.initializePawns(playersCount, items);
85              board = boardInitializer.initializeBoard(playersCount, items, pawns);
86              players = playersInitializer.initializePlayers(playersCount, items, pawns);
87              
88              turnLogic = new TurnLogic();
89              turnLogic.setStartState(getInitialState(board, players, turnLogic));
90              
91              game = new Game(turnLogic);
92              
93              return true;
94          } else 
95              return false;
96      } 
97      
98      /**
99       * Adds the new active player.
100      *
101      * @param nick the nick
102      * @param currentTurnHandler the current turn handler
103      * @param otherTurnHandler the other turn handler
104      * @throws PlayerWithGivenNickAlreadyExistsException the player with given nick already exists exception
105      */
106     public final void addNewActivePlayer(String nick, RemoteTurnHandlable currentTurnHandler, RemoteTurnHandlable otherTurnHandler) 
107                             throws PlayerWithGivenNickAlreadyExistsException{
108         for(Player p : players) {
109             if(p.getNick() == null) {
110                 p.setCurrentTurnHandler(currentTurnHandler);
111                 p.setOtherTurnHandler(otherTurnHandler);
112                 p.setNick(nick);
113                 return;
114             } else if(p.getNick().equals(nick))
115                 throw new PlayerWithGivenNickAlreadyExistsException();
116         }
117     }
118     
119     /**
120      * Removes the active player.
121      *
122      * @param nick the nick
123      */
124     public final void removeActivePlayer(String nick) {
125         for(Player p : players) {
126             if(p.getNick() == null) {
127                 p.setCurrentTurnHandler(null);
128                 p.setOtherTurnHandler(null);
129                 p.setNick(null);
130                 return;
131             }
132         }
133     }
134     
135     /**
136      * If all players ready.
137      *
138      * @return true, if successful
139      */
140     public final boolean ifAllPlayersReady() {
141         for(Player p : players) {
142             if(p.getNick() == null) {
143                 return false;
144             }
145         }
146         return true;
147     }
