30      @Test
31      public void testFieldConstructors() {
32          Field f = new Field(0);
33          List<Field> neighbours = new LinkedList<Field>();
34          List<Pawn> pawns = new LinkedList<Pawn>();
35          Map<String, Item> items= new HashMap<String, Item>();
36          pawns.add(new Pawn(0));
37          items.put("name",new Item(0));
38          
39          f.setGraphicsID(0);
40          f.setNeighbourFields(neighbours);
41          f.setPawns(pawns);
42          f.setItems(items);
43          
44          Field cons1 = new Field(0,0);
45          assertEquals(cons1.getID(), 0);
46          assertEquals(cons1.getGraphicsID(), 0);
47          assertEquals(cons1.getNeighbourFields(), f.getNeighbourFields());
48          
49          Field cons2 = new Field(0,0,pawns);
50          assertEquals(cons2.getID(), 0);
51          assertEquals(cons2.getGraphicsID(), 0);
52          assertEquals(cons2.getFieldsPawns(), pawns);
53          assertEquals(cons2.getNeighbourFields(), f.getNeighbourFields());
54          
55          Field cons21 = new Field(0,0,null);
56          assertEquals(cons21.getID(), 0);
57          assertEquals(cons21.getGraphicsID(), 0);
58          assertEquals(cons21.getFieldsPawns(), new LinkedList<Pawn>());
59          assertEquals(cons21.getNeighbourFields(), f.getNeighbourFields());
60          
61          Field cons3 = new Field(0,0,pawns,items);
62          assertEquals(cons3.getID(), 0);
63          assertEquals(cons3.getGraphicsID(), 0);
64          assertEquals(cons3.getFieldsPawns(), f.getFieldsPawns());
65          assertEquals(cons3.getItems(), f.getItems());
66          assertEquals(cons3.getNeighbourFields(), f.getNeighbourFields());
67          
68          Field cons31 = new Field(1,0,null,null);
69          assertEquals(cons31.getID(), 1);
70          assertEquals(cons31.getGraphicsID(), 0);
71          assertEquals(cons31.getFieldsPawns(), new LinkedList<Pawn>());
72          assertEquals(cons31.getItems(), new HashMap<String, Item>());
73          assertEquals(cons31.getNeighbourFields(), f.getNeighbourFields());
74          
75      }
