44  class StartState extends State {
45  
46      public StartState(Board board, Player currentPlayer,
47              List<Player> allPlayers, StateChangable stateChangable) {
48          super(board, currentPlayer, allPlayers, stateChangable);
49      }
50  
51      @Override
52      public Context getContext() {
53          return new Context("StartGame", board, super.currentPlayer , allPlayers) ;
54      }
55  
56      @Override
57      public void makeTurn(TurnResult currTurnResult,
58              Map<Player, TurnResult> otherTurnResults) {
59          Item item = new Item(0);
60          item.setAttribute("Amount", "300");
61          currentPlayer.setItem("Money", item );
62          BoardHelper boardHelper = new BoardHelper( board ) ;
63          try {
64              boardHelper.movePlayerFirstPawnByNum( currentPlayer, Integer.parseInt( currTurnResult.getProperty("Move") ) );
65          } catch (NumberFormatException e) {
66              // TODO Auto-generated catch block
67              e.printStackTrace();
68          } catch (NotPawnFoundException e) {
69              // TODO Auto-generated catch block
70              e.printStackTrace();
71          } catch (NotAvalaibleMoveException e) {
72              // TODO Auto-generated catch block
73              e.printStackTrace();
74          } catch (UnsupportedMoveNumberException e) {
75              // TODO Auto-generated catch block
76              e.printStackTrace();
77          } catch (PlayerNotHasPawnsException e) {
78              // TODO Auto-generated catch block
79              e.printStackTrace();
80          } catch (FieldNotHasNeihgboursException e) {
81              // TODO Auto-generated catch block
82              e.printStackTrace();
83          } 
84          currentPlayer = StateHelper.getNextPlayer(currentPlayer, allPlayers);
85          stateChangable.changeState( new GameEnd(board,currentPlayer,allPlayers, stateChangable) );
86      }
87  
88      @Override
89      public boolean gameEnded() {
90          // TODO Auto-generated method stub
91          return false;
92      }
93  }
