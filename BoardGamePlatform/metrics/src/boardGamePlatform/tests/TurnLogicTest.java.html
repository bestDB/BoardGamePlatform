94      @Test
95      public void test() throws UnsupportedBoardSizeException {
96          TurnLogic turnLogic = new TurnLogic();
97          
98          Player player1 = new Player(1);
99          Player player2 = new Player(2);
100         Player player3 = new Player(3);
101         Player player4 = new Player(4);
102         
103         Player currentPlayer = player2;
104         List<Player> allPlayers = new LinkedList<Player>();
105         for (Player player : new Player[] { player1, player2, player3 , player4 } )
106             allPlayers.add( player );
107         
108         Board board = new Board();
109         BoardBuilder boardBuilder = new BoardBuilder(board);
110         boardBuilder.makeSimpleLinearBoard(5);
111         
112         State startState = new StartGameState(board, currentPlayer, allPlayers, turnLogic);
113         turnLogic.setStartState(startState);
114         
115         Context firstContext = turnLogic.getContext();
116         assertEquals("StartGame", firstContext.getName() );
117         assertEquals( player2 , turnLogic.getCurrentPlayer() );
118         
119         turnLogic.makeTurn(null, null);
120         
121         Context secondContext = turnLogic.getContext();
122         assertEquals("PlayerMove", secondContext.getName() );
123         assertEquals( player3 , turnLogic.getCurrentPlayer() );
124     }
