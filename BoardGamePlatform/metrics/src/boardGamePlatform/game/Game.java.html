15  /**
16   * Glowna klasa zarzadzajaca calym przebiegiem rozgrywki. Do poprawnego dzialania wymaga podania
17   * logiki gry, ktora ma obslugiwac oraz pelnej listy graczy bioracych udzial w rozgrywce. 
18   */
19  public class Game {
20      
21      /** Logika rozgrywki. */
22      private TurnLogic turnLogic;
23      
24      /** Lista graczy bioracych udzial w rozgrywce. */
25      private List<Player> players;
26      
27      /**
28       * Instancjonuje nowy obiekt gry.
29       *
30       * @param turnLogic logika rozgrywki
31       */
32      public Game(TurnLogic turnLogic) {
33          this.turnLogic = turnLogic;
34      }
35      
36      /**
37       * Rozpoczyna rogrywke.
38       */
39      public void start() throws PlayerNotRespondException{
40          while( !turnLogic.gameEnded() ) {
41              Player currPlayer = turnLogic.getCurrentPlayer();
42              Context context = turnLogic.getContext();
43              TurnResult currTurnResult = null;
44              
45              Map<Player,TurnResult> otherTurnResult = new HashMap<Player,TurnResult>();
46              for (Player p : turnLogic.getAllPlayers() )
47                  if ( p != currPlayer)
48                      otherTurnResult.put(p, p.makeOtherTurn(context) );
49              
50              currTurnResult = currPlayer.makeCurrentTurn(context);
51              turnLogic.makeTurn(currTurnResult, otherTurnResult); 
52          }
53      }
54      
55      /**
56       * Zwraca obecna logike gry.
57       *
58       * @return logika gry
59       */
60      public TurnLogic getTurnLogic() {
61          return turnLogic;
62      }
63  }
