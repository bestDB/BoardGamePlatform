4   package boardGamePlatform.graphics;
5   
6   import java.util.HashMap;
7   import java.util.Map;
8   
9   // TODO: Auto-generated Javadoc
10  /**
11   * The Class Graphics.
12   */
13  public class Graphics {
14      
15      /** The drawables. */
16      private Map<String, Drawable> drawables;
17      
18      /**
19       * Instantiates a new graphics.
20       */
21      public Graphics() {
22          drawables = new HashMap<String, Drawable>();
23      }
24      
25      /**
26       * Draw.
27       *
28       * @param obj the obj
29       */
30      public void draw(Object obj) {
31          drawables.get(Drawable.getGID(obj)).draw();
32      }
33      
34      /**
35       * Adds the drawable.
36       *
37       * @param drawable the drawable
38       */
39      public void addDrawable(Drawable drawable) {
40          drawables.put(drawable.getGID(), drawable);
41      }
42      
43      /**
44       * Gets the drawable.
45       *
46       * @param obj the obj
47       * @return the drawable
48       */
49      public Drawable getDrawable(Object obj) {
50          String gid = Drawable.getGID(obj);
51          if(!drawables.containsKey(gid))
52              return null;
53          return drawables.get(gid);
54      }
55  }
