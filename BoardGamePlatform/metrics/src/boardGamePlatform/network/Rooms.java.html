5   
6   import java.rmi.RemoteException;
7   import java.rmi.server.UnicastRemoteObject;
8   import java.util.HashMap;
9   import java.util.List;
10  import java.util.Map;
11  
12  import boardGamePlatform.game.TurnHandler;
13  import boardGamePlatform.gameInitializers.GameFactory;
14  import boardGamePlatform.gameInitializers.Initializer;
15  import boardGamePlatform.platformExceptions.RoomMakingException;
16

18  /**
19   * The Class Rooms.
20   */
21  public class Rooms implements RoomsManager {
22      
23      /** The rooms. */
24      private Map<String, Room > rooms;
25      
26      /** The game factory. */ 
27      private GameFactory gameFactory ;
28      
29      
30      /**
31       * Instantiates a new rooms.
32       *
33       * @param gameFactory the game factory
34       * @throws RemoteException the remote exception
35       */
36      public Rooms(GameFactory gameFactory) throws RemoteException {
37          rooms = new HashMap<String, Room>(); 
38          this.gameFactory = gameFactory;
39      }
40      
41      /* (non-Javadoc)
42       * @see boardGamePlatform.network.RoomsManager#makeRoom()
43       */
44      @Override
45      public Managable makeRoom(String roomName, int playersCount, RemoteTurnHandlable currentTurnHandler,
46              RemoteTurnHandlable otherTurnHandler, GameHasEndedCallback gameHasEndedCallback, String nick) throws RemoteException,RoomMakingException {
47          Room room = new Room(new Initializer(gameFactory) , playersCount, currentTurnHandler, otherTurnHandler,
48                  gameHasEndedCallback, nick);
49          try {
50              Managable roomManagable = (Managable) UnicastRemoteObject.exportObject( room , 0);
51              rooms.put(roomName, room);
52              return roomManagable;
53          } catch(RemoteException remoteException) { 
54              throw new RoomMakingException();
55          }
56          
57      }
58      
59      /* (non-Javadoc)
60       * @see boardGamePlatform.network.RoomsManager#joinFirstAvailableRoom(java.lang.String, boardGamePlatform.game.TurnHandler, boardGamePlatform.game.TurnHandler, boardGamePlatform.network.Client)
61       */
62      @Override
63      public Accessible joinFirstAvailableRoom(String nick, RemoteTurnHandlable currTurnHandler, RemoteTurnHandlable otherTurnHandler, GameHasEndedCallback gameHasEndedCallback) throws RemoteException {
64          for (String room : rooms.keySet() ) {
65              if ( rooms.get( room ).isJoinable() )
66                  rooms.get(room).join(nick, currTurnHandler, otherTurnHandler, gameHasEndedCallback);
67                  return rooms.get( room );
68          }
69          return null;
70      }

72  }
