50      /* (non-Javadoc)
51       * @see boardGamePlatform.game.State#makeTurn(boardGamePlatform.game.TurnResult, java.util.Map)
52       */
53      @Override
54      public void makeTurn(TurnResult currTurnResult,
55          Map<Player, TurnResult> otherTurnResults) {
56          BoardHelper boardH = new BoardHelper(board);
57          try {
58              Field currField = boardH.getPlayersFirstPawnField(currentPlayer);
59              if((Integer.parseInt(currField.getItem("House").getAttribute("Amount")) == 0)
60                      && ( Integer.parseInt(currentPlayer.getItem("Money").getAttribute("Amount"))>= 150 ) ) {
61                  Map<String, String> attrs = new HashMap<String, String>();
62                  attrs.put("What", "House");
63                  stateChangable.changeState(new BuyingItem(board, currentPlayer, allPlayers, stateChangable, attrs));
64                  return;
65              } else {
66                  currentPlayer = StateHelper.getNextPlayer(currentPlayer, allPlayers);
67                  stateChangable.changeState(new PlayerMove(board, currentPlayer, allPlayers, stateChangable));
68                  return;
69              }
70                  
71          } catch (Exception e) {
72              e.printStackTrace();
73              return;
74          }
75      }
