169     /**
170      * Test buying field bankrupt.
171      *
172      * @throws NotPawnFoundException the not pawn found exception
173      * @throws PlayerNotHasPawnsException the player not has pawns exception
174      */
175     @Test
176     public void testBuyingFieldBankrupt() throws NotPawnFoundException, PlayerNotHasPawnsException {
177         BoardHelper boardHelper = new BoardHelper( board );
178         turnLogic.setStartState( new StartGame( board,one , new LinkedList<Player>( Arrays.asList(one,two) ), turnLogic ) );
179         
180         Map<Player,TurnResult> results = new HashMap<Player,TurnResult>();
181         results.put( two , new TurnResult() );
182         turnLogic.makeTurn(new TurnResult() , results );
183         
184         
185         assertEquals( "PlayerMove", turnLogic.getContext().getName() );
186         turnLogic.makeTurn( new TurnResult("Move","3") , results);
187         assertEquals("AtNobodysField", turnLogic.getContext().getName() );
188         assertEquals( 3 , boardHelper.getPlayersFirstPawnField(one).getID() );
189         assertEquals( 0 , boardHelper.getPlayersFirstPawnField(two).getID() );
190         
191         turnLogic.makeTurn(new TurnResult() , results);
192         assertEquals( "BuyingItem" , turnLogic.getContext().getName() );
193         
194         turnLogic.makeTurn( new TurnResult("Decision","Yes" ) , results);
195         assertEquals( "PlayerBought" , turnLogic.getContext().getName() );
196         turnLogic.makeTurn(new TurnResult() , results);
197         assertEquals( "PlayerMove" , turnLogic.getContext().getName() );
198         assertEquals( "900" , one.getItem("Money").getAttribute("Amount") );
199         assertEquals( "1000" , two.getItem("Money").getAttribute("Amount") );
200         assertEquals( "1" , boardHelper.getPlayersFirstPawnField(one).getAttribute("Owner") );
201         assertEquals( two , turnLogic.getCurrentPlayer() );
202         
203         two.getItem("Money").setAttribute("Amount", "90" );
204         
205         turnLogic.makeTurn(new TurnResult("Move","3") , results);
206         assertEquals( "AtEnemyField" , turnLogic.getContext().getName() );
207         turnLogic.makeTurn( new TurnResult() , results );
208         assertEquals( "PlayerBankrupt" , turnLogic.getContext().getName() );
209         turnLogic.makeTurn( new TurnResult() , results );
210         assertEquals("GameEnded" , turnLogic.getContext().getName() );
211         assertFalse( turnLogic.getAllPlayers().contains(two) ); 
212     }
