184      * Test.

186      * @throws NotPawnFoundException the not pawn found exception
187      * @throws PlayerNotHasPawnsException the player not has pawns exception
188      */
189     @Test
190     public void test() throws NotPawnFoundException, PlayerNotHasPawnsException {
191         Initializer monopolyInitializer = new Initializer( new MonopolyGameFactory() );
192         
193         assertEquals(true, monopolyInitializer.initialize(2));
194         assertEquals(false, monopolyInitializer.ifAllPlayersReady());
195         
196         try {
197             monopolyInitializer.addNewActivePlayer("Zbychu", new RemoteTurnHandler(new TurnHandler() ), new RemoteTurnHandler( new TurnHandler() ));
198             assertEquals(false, monopolyInitializer.ifAllPlayersReady());
199             monopolyInitializer.addNewActivePlayer("Stachu", new RemoteTurnHandler( new TurnHandler() ), new RemoteTurnHandler( new TurnHandler() ));
200         } catch (PlayerWithGivenNickAlreadyExistsException e) {
201             // TODO Auto-generated catch block
202             e.printStackTrace();
203         }
204         
205         assertEquals(true, monopolyInitializer.ifAllPlayersReady());
206         
207         BoardHelper boardHelper = new BoardHelper(monopolyInitializer.getBoard());
208         List<Player> players = monopolyInitializer.getPlayers();
209         
210         assertEquals(0, boardHelper.getPlayersFirstPawnField(players.get(0)).getID());
211         assertEquals(1, boardHelper.getPlayersFirstPawnField(players.get(1)).getID());
212         
213         assertNotNull(players.get(0).getCurrentTurnHandler());
214         assertNotNull(players.get(1).getCurrentTurnHandler());
215         
216         assertNotNull(players.get(0).getOtherTurnHandler());
217         assertNotNull(players.get(1).getOtherTurnHandler());
218         
219         assertEquals("1000",players.get(0).getItem("Money").getAttribute("Value"));
220         
221         assertEquals("StartGame",monopolyInitializer.getTurnLogic().getContext().getName());
222         
223         
224         assertEquals(false, monopolyInitializer.initialize(3));
225         
226         System.out.println(monopolyInitializer.getBoard());
